// ── Closing calc ──
function calcClosing() {
  const opening = parseFloat(document.getElementById('s1-opening').value) || 0;
  const revenue = parseFloat(document.getElementById('f-revenue').value) || 0;
  const card    = parseFloat(document.getElementById('f-card').value) || 0;
  const inkass  = parseFloat(document.getElementById('f-inkass').value) || 0;
  const closing = parseFloat(document.getElementById('f-closing').value) || 0;
  const expenses = window._totalExpenses || 0;

  const calculated = opening + revenue - expenses - card - inkass;
  const diff = closing - calculated;

  document.getElementById('r-calc').textContent   = revenue > 0 ? Math.round(calculated).toLocaleString('ru') + ' ₽' : '—';
  document.getElementById('r-actual').textContent = closing > 0 ? Math.round(closing).toLocaleString('ru') + ' ₽' : '—';

  const ned = document.getElementById('v-ned');
  const izl = document.getElementById('v-izl');
  const diffEl = document.getElementById('r-diff');

  if (revenue > 0 && closing > 0) {
    const abs = Math.abs(Math.round(diff)).toLocaleString('ru') + ' ₽';
    if (diff < 0) {
      diffEl.textContent = '−' + abs; diffEl.style.color = '#e05555';
      ned.classList.add('show'); izl.classList.remove('show');
      document.getElementById('v-ned-amt').textContent = abs + ' не хватает в кассе';
    } else if (diff > 0) {
      diffEl.textContent = '+' + abs; diffEl.style.color = '#4DC87A';
      izl.classList.add('show'); ned.classList.remove('show');
      document.getElementById('v-izl-amt').textContent = abs + ' сверх расчёта';
    } else {
      diffEl.textContent = '0 ₽ — всё сходится'; diffEl.style.color = '#4DC87A';
      ned.classList.remove('show'); izl.classList.remove('show');
    }
  } else {
    diffEl.textContent = '—'; diffEl.style.color = '#9FD4B8';
    ned.classList.remove('show'); izl.classList.remove('show');
  }
}

// ── Compress image ──
function compressImage(file, maxWidth = 1280, quality = 0.8) {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onerror = () => reject(new Error('Не удалось прочитать файл'));
    reader.onload = e => {
      const img = new Image();
      img.onerror = () => reject(new Error('Не удалось загрузить изображение'));
      img.onload = () => {
        let { width, height } = img;
        if (width > maxWidth) { height = Math.round(height * maxWidth / width); width = maxWidth; }
        const canvas = document.createElement('canvas');
        canvas.width = width; canvas.height = height;
        canvas.getContext('2d').drawImage(img, 0, 0, width, height);
        resolve(canvas.toDataURL('image/jpeg', quality).split(',')[1]);
      };
      img.src = e.target.result;
    };
    reader.readAsDataURL(file);
  });
}

// ── Collect salary ──
function collectSalary() {
  const map = {};
  let hasError = false;

  document.querySelectorAll('#exp-salary .salary-item').forEach(item => {
    const title = item.querySelector('.salary-title').textContent.trim();
    let total = 0;
    const employeeNames = [];

    item.querySelectorAll('.salary-line').forEach(line => {
      const amount = parseFloat(line.querySelector('input[type=number]')?.value) || 0;
      if (amount > 0) {
        total += amount;
        if (title === 'Другой сотрудник') {
          const nameInput = line.querySelector('.other-employee-name');
          const name = nameInput?.value.trim() || '';
          if (!name) { if (nameInput) nameInput.style.borderColor = '#e05555'; hasError = true; }
          else { if (nameInput) nameInput.style.borderColor = '#3A7A4E'; employeeNames.push(name); }
        }
      }
    });

    if (total > 0) {
      if (title === 'Другой сотрудник' && employeeNames.length > 0) {
        map[title + ' (' + employeeNames.join(', ') + ')'] = total;
      } else if (title !== 'Другой сотрудник') {
        map[title] = total;
      }
    }
  });

  if (hasError) {
    alert('Пожалуйста, укажите ФИО для всех сотрудников в категории "Другой сотрудник"');
    return null;
  }
  return map;
}

// ── Submit report ──
async function submitReport() {
  const validFiles = photoFiles.filter(f => f !== null);
  if (validFiles.length === 0) {
    alert('Пожалуйста, загрузите фото накладных!\n\nБез фото отчёт не может быть отправлен.');
    return;
  }
  if (!document.getElementById('f-revenue').value) { alert('Укажите выручку'); return; }
  if (!document.getElementById('f-closing').value) { alert('Укажите остаток на конец дня'); return; }

  const submitBtn = document.getElementById('submit-btn');
  submitBtn.disabled = true;
  submitBtn.textContent = 'Подготовка данных...';

  const dateValue = document.getElementById('s1-date').value;
  // Всегда приводим к ISO для бэкенда
  let backendDate, displayDate;
  if (dateValue.includes('-')) {
    backendDate = dateValue;
    displayDate = formatDate(dateValue);
  } else {
    const p = dateValue.split('.');
    backendDate = p[2] + '-' + p[1] + '-' + p[0];
    displayDate = dateValue;
  }

  const opening  = parseFloat(document.getElementById('s1-opening').value) || 0;
  const revenue  = parseFloat(document.getElementById('f-revenue').value) || 0;
  const card     = parseFloat(document.getElementById('f-card').value) || 0;
  const inkass   = parseFloat(document.getElementById('f-inkass').value) || 0;
  const closing  = parseFloat(document.getElementById('f-closing').value) || 0;
  const expenses = window._totalExpenses || 0;
  const calc     = opening + revenue - expenses - card - inkass;
  const diff     = closing - calc;

  const salary = collectSalary();
  if (salary === null) {
    submitBtn.disabled = false;
    submitBtn.textContent = 'Отправить отчёт ↗';
    return;
  }

  const invoices = collectInvoices();
  const totalInvoices = invoices['__total'] || 0;
  delete invoices['__total'];

  const data = {
    type:           'shift',          // ← ВАЖНО: явно указываем тип
    date:           backendDate,
    displayDate:    displayDate,
    name:           document.getElementById('s1-name').value.trim(),
    location:       document.getElementById('s1-loc').value,
    opening:        opening,
    openingComment: document.getElementById('s1-comment').value,
    suppliers:      collectSuppliers(),
    otherPayments:  collectOtherPayments(),
    salary:         salary,
    totalExpenses:  expenses,
    invoices:       invoices,
    totalInvoices:  totalInvoices,
    revenue:        revenue,
    card:           card,
    inkass:         inkass,
    safe:           parseFloat(document.getElementById('f-safe').value) || 0,
    closing:        closing,
    calcBalance:    Math.round(calc),
    diff:           Math.round(diff),
    verdict:        diff < 0 ? 'Недостача' : diff > 0 ? 'Излишек' : 'Сходится',
    photoCount:     validFiles.length,
    comment:        document.getElementById('f-comment').value
  };

  try {
    // Шаг 1: основные данные
    submitBtn.textContent = 'Отправка отчёта...';
    const response = await fetch(APPS_SCRIPT_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'text/plain;charset=utf-8' },
      body: JSON.stringify(data)
    });

    let result;
    try {
      result = await response.json();
    } catch (e) {
      throw new Error('Сервер вернул некорректный ответ');
    }

    if (!result.ok) throw new Error(result.error || 'Ошибка сервера');

    const reportId = result.reportId || (backendDate + '_' + data.name);

    // Шаг 2: фото по одному
    for (let i = 0; i < photoFiles.length; i++) {
      const file = photoFiles[i];
      if (!file) continue;

      submitBtn.textContent = `Отправка фото ${i + 1}/${validFiles.length}...`;

      try {
        const base64 = await compressImage(file, 800, 0.6);
        await fetch(APPS_SCRIPT_URL, {
          method: 'POST',
          headers: { 'Content-Type': 'text/plain;charset=utf-8' },
          body: JSON.stringify({
            type:        'photo',
            reportId:    reportId,
            photoIndex:  i,
            photoBase64: base64,
            location:    data.location,
            name:        data.name,
            date:        displayDate
          })
        });
      } catch (photoErr) {
        console.error('Ошибка фото ' + (i + 1) + ':', photoErr);
      }

      await new Promise(r => setTimeout(r, 300));
    }

    go('screen-success');

  } catch (err) {
    console.error('Ошибка отправки:', err);
    alert('Ошибка отправки: ' + err.message);
    submitBtn.disabled = false;
    submitBtn.textContent = 'Отправить отчёт ↗';
  }
}

function populatePointAccounts() {
  document.querySelectorAll('.po-account').forEach(select => {
    const current = select.value;
    select.innerHTML = '<option value="" disabled selected>Выберите...</option>';
    POINT_ACCOUNTS.forEach(account => {
      const opt = document.createElement('option');
      opt.value = account;
      opt.textContent = account;
      select.appendChild(opt);
    });
    if (current) select.value = current;
  });
}

function populatePointAnalytics() {
  const list = document.getElementById('po-analytics-list');
  if (!list) return;
  list.innerHTML = '';
  POINT_ANALYTICS.forEach(item => {
    const opt = document.createElement('option');
    opt.value = item;
    list.appendChild(opt);
  });
}

function setDefaultPointDate() {
  const dateInput = document.getElementById('po-date');
  if (!dateInput) return;
  if (!dateInput.value) {
    dateInput.value = new Date().toISOString().slice(0, 10);
  }
}

function setPointDots(step) {
  const dots = {
    location: document.getElementById('po-dot-location'),
    type: document.getElementById('po-dot-type'),
    form: document.getElementById('po-dot-form')
  };
  Object.values(dots).forEach(dot => dot && dot.classList.remove('active'));
  if (dots[step]) dots[step].classList.add('active');
}

function showPointOperationType() {
  const type = document.getElementById('po-type').value;

    document.getElementById('po-transfer-points-from-location').value = pointOperationState.location || document.getElementById('po-location').value;

}

function getPointCommonData() {
  const location = document.getElementById('po-location').value;
  const operationType = document.getElementById('po-type').value;
  const date = document.getElementById('po-date').value;

  if (!location) throw new Error('Выберите точку');
  if (!date) throw new Error('Укажите дату');
  if (!operationType) throw new Error('Выберите тип операции');

  pointOperationState.location = location;
  pointOperationState.lastDate = date;
  pointOperationState.lastOperationType = operationType;

  return {
    type: 'point-operation',
    location,
    operationType,
    date,
    timestamp: new Date().toISOString()
  };
}

function getNumberValue(id, fieldName) {
  const value = parseFloat(document.getElementById(id).value);
  if (!value || value <= 0) throw new Error('Укажите сумму: ' + fieldName);
  return value;
}

function getRequiredValue(id, fieldName) {
  const el = document.getElementById(id);
  const value = el ? String(el.value || '').trim() : '';
  if (!value) throw new Error('Заполните поле: ' + fieldName);
  return value;
}

function collectPointOperationData() {
  const data = getPointCommonData();

  if (data.operationType === 'expense') {
    data.account = getRequiredValue('po-expense-account', 'Счет списания');
    data.amount = getNumberValue('po-expense-amount', 'Сумма расхода');
    data.analytics = getRequiredValue('po-expense-analytics', 'Аналитика');
    data.details = document.getElementById('po-expense-details').value.trim();
    data.comment = document.getElementById('po-expense-comment').value.trim();
    data.sign = '-';
  }

  else if (data.operationType === 'income') {
    data.account = getRequiredValue('po-income-account', 'Счет зачисления');
    data.amount = getNumberValue('po-income-amount', 'Сумма поступления');
    data.analytics = document.getElementById('po-income-analytics').value.trim() || 'Поступление средств';
    data.details = document.getElementById('po-income-details').value.trim();
    data.comment = document.getElementById('po-income-comment').value.trim();
    data.sign = '+';
  }

  else if (data.operationType === 'transfer_accounts') {
    data.fromAccount = getRequiredValue('po-transfer-accounts-from', 'С какого счета');
    data.toAccount = getRequiredValue('po-transfer-accounts-to', 'На какой счет');
    data.amount = getNumberValue('po-transfer-accounts-amount', 'Сумма перемещения');
    data.analytics = 'Перемещение средств';
    data.comment = document.getElementById('po-transfer-accounts-comment').value.trim();
    if (data.fromAccount === data.toAccount) throw new Error('Счета списания и зачисления не должны совпадать');
  }

  else if (data.operationType === 'transfer_points') {
    data.fromLocation = data.location;
    data.toLocation = getRequiredValue('po-transfer-points-to-location', 'Точка-получатель');
    data.fromAccount = getRequiredValue('po-transfer-points-from-account', 'Счет списания');
    data.toAccount = 'перемещенные средства';
    data.amount = getNumberValue('po-transfer-points-amount', 'Сумма перемещения');
    data.analytics = 'Перемещение средств';
    data.comment = document.getElementById('po-transfer-points-comment').value.trim();
    if (data.fromLocation === data.toLocation) throw new Error('Точка-отправитель и точка-получатель не должны совпадать');
  }

  else if (data.operationType === 'cash_fact') {
    data.factPeriod = getRequiredValue('po-cash-fact-period', 'Факт на');
    data.account = getRequiredValue('po-cash-fact-account', 'Счет');
    data.amount = getNumberValue('po-cash-fact-amount', 'Фактическая сумма');
    data.analytics = 'Факт средств';
    data.details = data.factPeriod;
    data.comment = document.getElementById('po-cash-fact-comment').value.trim();
  }

  else if (data.operationType === 'debt') {
    data.account = 'долг поставщикам';
    data.amount = getNumberValue('po-debt-amount', 'Сумма долга');
    data.analytics = document.getElementById('po-debt-analytics').value.trim() || 'Долг';
    data.details = getRequiredValue('po-debt-details', 'Поставщик / уточнение');
    data.comment = document.getElementById('po-debt-comment').value.trim();
    data.sign = '+';
  }

  else if (data.operationType === 'debt_payment') {
    data.fromAccount = getRequiredValue('po-debt-payment-account', 'С какого счета оплачено');
    data.debtAccount = 'долг поставщикам';
    data.amount = getNumberValue('po-debt-payment-amount', 'Сумма оплаты долга');
    data.analytics = document.getElementById('po-debt-payment-analytics').value.trim() || 'Оплата долга';
    data.details = getRequiredValue('po-debt-payment-details', 'Поставщик / уточнение');
    data.comment = document.getElementById('po-debt-payment-comment').value.trim();
  }

  else if (data.operationType === 'owner_withdrawal') {
    data.account = getRequiredValue('po-owner-withdrawal-account', 'Счет списания');
    data.person = getRequiredValue('po-owner-withdrawal-person', 'Кто забрал');
    data.amount = getNumberValue('po-owner-withdrawal-amount', 'Сумма вывода');
    data.analytics = 'Вывод средств';
    data.details = data.person;
    data.comment = document.getElementById('po-owner-withdrawal-comment').value.trim();
    data.sign = '-';
  }

  else if (data.operationType === 'cash_difference') {
    data.account = getRequiredValue('po-cash-difference-account', 'Счет');
    data.differenceKind = getRequiredValue('po-cash-difference-kind', 'Тип расхождения');
    data.amount = getNumberValue('po-cash-difference-amount', 'Сумма расхождения');
    data.analytics = data.differenceKind;
    data.details = 'вручную';
    data.comment = document.getElementById('po-cash-difference-comment').value.trim();
    data.sign = data.differenceKind === 'Излишек' ? '+' : '-';
  }

  return data;
}

async function submitPointOperation() {
  const activeForm = document.querySelector('.po-form[style*="display: block"]');
  const submitBtn = activeForm ? activeForm.querySelector('.btn-primary') : null;

  try {
    const data = collectPointOperationData();

    if (submitBtn) {
      submitBtn.disabled = true;
      submitBtn.textContent = 'Отправка...';
    }

    const response = await fetch(APP_SCRIPT_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'text/plain;charset=utf-8' },
      body: JSON.stringify(data)
    });

    let result;
    try {
      result = await response.json();
    } catch (e) {
      throw new Error('Сервер вернул некорректный ответ');
    }

    if (!result.ok) throw new Error(result.error || 'Ошибка сервера');

    const typeLabel = getPointOperationTypeLabel(data.operationType);
    document.getElementById('po-success-sub').innerHTML = `${typeLabel}<br>${data.location} · ${formatDate(data.date)}`;
    go('screen-off-cash-success');
    clearPointOperationFields(true);

  } catch (err) {
    alert(err.message || 'Ошибка отправки');
  } finally {
    if (submitBtn) {
      submitBtn.disabled = false;
      submitBtn.textContent = 'Отправить операцию ↗';
    }
  }
}

function getPointOperationTypeLabel(type) {
  const labels = {
    expense: 'Расход',
    income: 'Поступление',
    transfer_accounts: 'Перемещение между счетами',
    transfer_points: 'Перемещение между точками',
    cash_fact: 'Факт средств',
    debt: 'Долг',
    debt_payment: 'Оплата долга',
    owner_withdrawal: 'Вывод средств',
    cash_difference: 'Расхождение кассы'
  };
  return labels[type] || 'Операция';
}

function clearPointOperationFields(keepLocationAndDate) {
  const savedLocation = pointOperationState.location || document.getElementById('po-location').value;
  const savedDate = pointOperationState.lastDate || document.getElementById('po-date').value;

  document.querySelectorAll('#screen-off-cash input, #screen-off-cash textarea').forEach(el => {
    if (el.id === 'po-date' && keepLocationAndDate) return;
    if (el.id === 'po-transfer-points-from-location' && keepLocationAndDate) return;
    if (el.type === 'text' || el.type === 'number' || el.tagName === 'TEXTAREA') {
      if (!el.readOnly) el.value = '';
    }
  });

  document.querySelectorAll('#screen-off-cash select').forEach(sel => {
    if (sel.id === 'po-location' && keepLocationAndDate) return;
    sel.selectedIndex = 0;
  });

  document.querySelectorAll('.po-form').forEach(form => form.style.display = 'none');
  document.getElementById('po-step-form').style.display = 'none';

  if (keepLocationAndDate) {
    document.getElementById('po-location').value = savedLocation;
    document.getElementById('po-date').value = savedDate;
    document.getElementById('po-selected-location').textContent = savedLocation || '—';
    document.getElementById('po-transfer-points-from-location').value = savedLocation || '';
    document.getElementById('po-step-operation').style.display = savedLocation ? 'block' : 'none';
    setPointDots(savedLocation ? 'type' : 'location');
  }
}

function resetPointOperationForm(keepLocation) {
  clearPointOperationFields(!!keepLocation);
  if (!keepLocation) {
    pointOperationState = { location: '', lastDate: '', lastOperationType: '' };
    document.getElementById('po-location').selectedIndex = 0;
    document.getElementById('po-step-operation').style.display = 'none';
    document.getElementById('po-step-form').style.display = 'none';
    setDefaultPointDate();
    setPointDots('location');
  }
}

  document.getElementById('po-date').value = dt;

}
